CMAKE_MINIMUM_REQUIRED(VERSION 2.8 FATAL_ERROR)
PROJECT(NanoBoyAdvance CXX)
SET(projectName NanoBoyAdvance)

FIND_PACKAGE(SDL 1.2 REQUIRED)
INCLUDE_DIRECTORIES(${SDL_INCLUDE_DIRS})
ADD_DEFINITIONS(${SDL_DEFINITIONS})

SET(SOURCE ${CMAKE_CURRENT_SOURCE_DIR}/main.cpp)
ADD_SUBDIRECTORY("${PROJECT_SOURCE_DIR}/common")
ADD_SUBDIRECTORY("${PROJECT_SOURCE_DIR}/core")
ADD_EXECUTABLE(nanoboyadvance main.cpp)
TARGET_COMPILE_FEATURES(nanoboyadvance PRIVATE cxx_range_for)
